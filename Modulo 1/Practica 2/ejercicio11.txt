PIC EQU 20H
EOI EQU 20H
N_F10 EQU 10

ORG 40
IP_F10 DW RUT_F10

ORG 1000H
RESULTADO DB ?
A DB "A"
Z DB "Z"
ORG 3000H
RUT_F10: PUSH AX
PUSH BX

MOV RESULTADO, DL
MOV BX, OFFSET RESULTADO
MOV AL, 1
INT 7

MOV AL, EOI
OUT EOI, AL ; PIC: registro EOI
POP BX
POP AX
IRET
ORG 2000H

CLI
MOV AL, 0FEH
OUT PIC+1, AL 
MOV AL, N_F10
OUT PIC+4, AL
STI

SET: MOV DL, A
LAZO: INC DL
CMP DL, Z
JS LAZO
JMP SET
INT 0
END
